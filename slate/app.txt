import requests
import time

def main():
    while True:
        try:
            response = requests.get('http://rest:9876')
            print(response.text)
        except requests.exceptions.RequestException as e:
            print(f"Request failed: {e}")
        time.sleep(5)

if __name__ == "__main__":
    main()
