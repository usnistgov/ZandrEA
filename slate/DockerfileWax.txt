# Per RChat - Most simple, no venv

# Use an official Python image as the base
FROM python:3.12

# Set the working directory to /app
WORKDIR /app

# Copy the requirements file
COPY requirements.txt .

# Install any dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Copy the application code
COPY . .

# Run the command to start the Python program
CMD ["python", "app.py"]


#----------------------------------------------------------------------
# Per RChat - Less simple , creates a venv in container

# Use an official Python image as the base
FROM python:3.12

LABEL maintainer="DAV <daniel.veronica@nist.gov>"

# Set the working directory to /app
WORKDIR /app

# Create a virtual environment
RUN python -m venv venv

# Activate the virtual environment
# [According to RChat, this set of PATH to include .venv/bin will persist through container restarts, so the venv does not have to be explicitly activated again. (DAV)]
ENV PATH="/app/venv/bin:$PATH"

# Copy the requirements file
COPY requirements.txt .

# Install dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Copy the application code
COPY . .

# Run the command to start the Python program
CMD ["python", "app.py"]
