// Implementation of "app" testing Experior (the microservice intended for experimental Python code)

#include "experior.hpp"

class CExperiorClient {

public:

   CExperiorClient( std::shared_ptr<grpc::Channel> channel )

                    : stub_( microservice::MakeComparison::NewStub( channel ) ) {

      // no constructor body
   }

   bool Is_x_bigger(float x, float y) {

      microservice::ComparisonInput input;
      input.set_x(x);
      input.set_y(y);

      microservice::MakeComparison reply;
      grpc::ClientContext context;
      grpc.Status status = stub_->Is_x_bigger( &context, input, &reply );

      if (status.ok()) {
         return reply.is_greater();
      } else {
         std::cerr << "RPC failed: " << status.error_message() << std::endl;
         return false; // Or handle error as appropriate
      }
   }

private:

    std::unique_ptr<microservice::MakeComparison::Stub> stub_;

};

int main() {
    std::string address("microservice:50051");
    auto channel = grpc::CreateChannel( address, grpc::InsecureChannelCredentials() );
    CExperiorClient client( channel );

    float x = 10.5;
    float y = 20.7;
    bool reply = client.Is_x_bigger( x, y );
    std::cout << "Is " << x << " greater than " << y << "? " << ( reply ? "Yes" : "No" ) << std::endl;

    return 0;
}
